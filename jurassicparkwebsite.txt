import React, { useState, useEffect } from 'react';
import { initializeApp } from 'firebase/app';
import { getAuth, signInWithCustomToken, signInAnonymously, onAuthStateChanged } from 'firebase/auth';
import { getFirestore, collection, addDoc, onSnapshot, deleteDoc, doc } from 'firebase/firestore';
import { 
  FileText, 
  Image as ImageIcon, 
  Upload, 
  FolderLock, 
  Database, 
  Search, 
  ShieldAlert,
  X,
  Trash2,
  Loader2,
  AlertTriangle,
  FileCode,
  Info,
  KeyRound,
  Eye,
  EyeOff,
  Fingerprint,
  Zap,
  Lock,
  ShieldCheck
} from 'lucide-react';

const _0x51a1 = "QTI5OQ=="; 
const _0x92b2 = "ZW50aXR5NDA0NjQyOQ=="; 

// --- PERSISTENT MASTER LOCK ---
// Paste your UID here (found in the sidebar Fingerprint) to lock deletion to your specific device.
const _0xMASTER_UID = ""; 

const decode = (str) => atob(str);

// --- FIREBASE INITIALIZATION ---
const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {};
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

export default function App() {
  const [documents, setDocuments] = useState([]);
  const [selectedDoc, setSelectedDoc] = useState(null);
  const [searchQuery, setSearchQuery] = useState("");
  const [isUploadOpen, setIsUploadOpen] = useState(false);
  const [viewMode, setViewMode] = useState('browse');
  const [user, setUser] = useState(null);
  const [isUploading, setIsUploading] = useState(false);
  const [errorMsg, setErrorMsg] = useState("");
  const [authInput, setAuthInput] = useState("");
  const [idInput, setIdInput] = useState("");
  const [showCode, setShowCode] = useState(false);
  const [revealUID, setRevealUID] = useState(false);
  const [purgeConfirmId, setPurgeConfirmId] = useState(null);
  
  const [isAdminSession, setIsAdminSession] = useState(false);
  const [terminalUnlocked, setTerminalUnlocked] = useState(false);

  // Verification Logic
  const isMasterUser = user && (_0xMASTER_UID === "" || user.uid === _0xMASTER_UID);
  const canModify = isAdminSession && user;
  const canDelete = canModify && isMasterUser;

  const [formData, setFormData] = useState({
    title: "",
    category: "InGen Files",
    type: "text", 
    content: "",
    classification: "Restricted"
  });
  const [selectedFile, setSelectedFile] = useState(null);

  useEffect(() => {
    const initAuth = async () => {
      try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
          await signInWithCustomToken(auth, __initial_auth_token);
        } else {
          await signInAnonymously(auth);
        }
      } catch (err) { console.error("Auth error:", err); }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, setUser);
    return () => unsubscribe();
  }, []);

  useEffect(() => {
    if (!user) return;
    const recordsRef = collection(db, 'artifacts', appId, 'public', 'data', 'ingen_records');
    const unsubscribe = onSnapshot(recordsRef, (snapshot) => {
      const fetchedRecords = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
      fetchedRecords.sort((a, b) => new Date(b.timestamp) - new Date(a.timestamp));
      setDocuments(fetchedRecords);
    }, (error) => {
      console.error("Firestore error:", error);
    });
    return () => unsubscribe();
  }, [user]);

  const handleAuthorize = () => {
    if (idInput === decode(_0x92b2) && authInput === decode(_0x51a1)) {
      setIsAdminSession(true);
      setErrorMsg("");
    } else {
      setErrorMsg("INVALID IDENTITY OR PROTOCOL CODE.");
      setIsAdminSession(false);
    }
  };

  const fileToBase64 = (file) => {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onload = () => resolve(reader.result);
      reader.onerror = (error) => reject(error);
    });
  };

  const handleUpload = async () => {
    if (!canModify) return;
    if (!formData.title) { setErrorMsg("TITLE REQUIRED."); return; }

    setIsUploading(true);
    setErrorMsg("");

    try {
      let fileData = "";
      if (formData.type !== 'text' && selectedFile) {
        if (selectedFile.size > 1000000) throw new Error("BUFFER OVERFLOW: 1MB LIMIT.");
        fileData = await fileToBase64(selectedFile);
      }

      const newRecord = {
        title: formData.title,
        category: formData.category,
        type: formData.type,
        content: formData.type === 'text' ? formData.content : "",
        classification: formData.classification,
        fileData: fileData,
        dateAdded: new Date().toLocaleDateString(),
        timestamp: new Date().toISOString(),
        authorUID: user.uid
      };

      const recordsRef = collection(db, 'artifacts', appId, 'public', 'data', 'ingen_records');
      await addDoc(recordsRef, newRecord);

      setFormData({ title: "", category: "InGen Files", type: "text", content: "", classification: "Restricted" });
      setSelectedFile(null);
      setIsUploadOpen(false);
    } catch (err) {
      setErrorMsg(err.message || "UPLINK FAILURE.");
    } finally {
      setIsUploading(false);
    }
  };

  const handleDelete = async (id) => {
    if (!canDelete) {
      setErrorMsg("PURGE DENIED: INSUFFICIENT CLEARANCE.");
      return;
    }
    try {
      await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'ingen_records', id));
      if (selectedDoc?.id === id) setSelectedDoc(null);
      setPurgeConfirmId(null);
    } catch (err) { console.error("PURGE ERROR:", err); }
  };

  const filteredDocs = documents.filter(doc => 
    doc.title.toLowerCase().includes(searchQuery.toLowerCase()) || 
    doc.category.toLowerCase().includes(searchQuery.toLowerCase())
  );

  return (
    <div className="min-h-screen bg-slate-100 text-slate-900 font-sans flex flex-col selection:bg-amber-200">
      <header className="bg-slate-900 text-slate-100 p-4 shadow-lg flex items-center justify-between border-b-2 border-amber-600">
        <div className="flex items-center space-x-3">
          <div 
            className="bg-amber-600 p-1.5 rounded-sm cursor-help active:scale-95 transition border border-amber-500/50"
            onClick={() => setTerminalUnlocked(!terminalUnlocked)}
          >
            <Database className="w-5 h-5 text-white" />
          </div>
          <div>
            <h1 className="text-lg font-bold tracking-tighter uppercase italic leading-none">InGen Archival System</h1>
            <p className="text-[10px] font-mono text-slate-400 tracking-wider mt-1">LORE PRESERVATION // PROJECT: EVILUTION</p>
          </div>
        </div>
        
        <div className="flex items-center space-x-4">
          <div className="relative">
            <Search className="w-4 h-4 absolute left-3 top-1/2 -translate-y-1/2 text-slate-500" />
            <input 
              type="text" 
              placeholder="Search Archives..." 
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
              className="bg-slate-800 text-sm rounded-sm pl-9 pr-4 py-1.5 focus:outline-none border border-slate-700 font-mono text-white w-64 transition-all focus:border-amber-500"
            />
          </div>
          
          {terminalUnlocked && (
            <button 
              onClick={() => setIsUploadOpen(true)} 
              className={`px-4 py-1.5 rounded-sm text-xs font-bold uppercase tracking-widest transition flex items-center space-x-2 shadow-lg border ${isAdminSession ? 'bg-amber-600 hover:bg-amber-700 text-white border-amber-500 animate-pulse' : 'bg-slate-700 text-slate-400 border-slate-600'}`}
            >
              {isAdminSession ? <Upload className="w-3 h-3" /> : <ShieldAlert className="w-3 h-3 text-amber-500" />}
              <span>{isAdminSession ? "Ingest" : "Authorize"}</span>
            </button>
          )}
        </div>
      </header>

      <div className="flex flex-1 overflow-hidden">
        {/* SIDEBAR */}
        <aside className="w-64 bg-slate-800 p-4 hidden md:flex flex-col border-r border-slate-700">
          <div className="space-y-6">
            <div>
              <h2 className="text-[10px] font-bold text-slate-500 uppercase tracking-widest mb-4 font-mono">Terminal Index</h2>
              <nav className="space-y-1">
                <button onClick={() => { setViewMode('browse'); setSearchQuery(''); }} className={`w-full flex items-center space-x-2 p-2.5 text-xs rounded transition font-mono uppercase ${viewMode === 'browse' ? 'bg-slate-700 text-white border-l-2 border-amber-500' : 'text-slate-400 hover:bg-slate-700 hover:text-white'}`}>
                  <FolderLock className="w-3 h-3 text-amber-600" />
                  <span>The Database</span>
                </button>
                <button onClick={() => setViewMode('disclaimer')} className={`w-full flex items-center space-x-2 p-2.5 text-xs rounded transition font-mono uppercase ${viewMode === 'disclaimer' ? 'bg-slate-700 text-white border-l-2 border-blue-500' : 'text-slate-400 hover:bg-slate-700 hover:text-white'}`}>
                  <Info className="w-3 h-3 text-blue-400" />
                  <span>InGen Policy</span>
                </button>
              </nav>
            </div>
          </div>
          
          <div className="mt-auto space-y-4">
            {terminalUnlocked && (
              <div className="p-3 bg-slate-900/50 rounded border border-slate-700 font-mono group">
                <div className="flex items-center justify-between mb-1">
                  <div className="flex items-center space-x-2 text-slate-500">
                    <Fingerprint className={`w-3 h-3 ${isAdminSession ? 'text-amber-500' : 'text-slate-600'}`} />
                    <span className="text-[8px] uppercase font-bold tracking-widest">Hash Auth</span>
                  </div>
                  <button onClick={() => setRevealUID(!revealUID)} className="opacity-0 group-hover:opacity-100 transition">
                    {revealUID ? <EyeOff className="w-2.5 h-2.5 text-slate-600" /> : <Eye className="w-2.5 h-2.5 text-slate-600" />}
                  </button>
                </div>
                {revealUID ? (
                  <p className="text-[7px] text-amber-600/70 break-all select-all leading-tight">{user?.uid}</p>
                ) : (
                  <p className="text-[8px] text-slate-600 italic">••••••••••••••••</p>
                )}
                <div className="mt-2 flex items-center space-x-1">
                  <Lock className={`w-2.5 h-2.5 ${canDelete ? 'text-amber-500' : 'text-slate-700'}`} />
                  <p className="text-[8px] text-slate-500 uppercase font-bold">
                    {isAdminSession ? (isMasterUser ? "Master Control" : "Admin Only") : "Public Read"}
                  </p>
                </div>
              </div>
            )}
            <div className="p-4 bg-slate-900 rounded border border-slate-700">
              <div className="flex items-center space-x-2 mb-2">
                <ShieldAlert className="w-3 h-3 text-amber-500" />
                <span className="text-[9px] font-mono text-amber-500 font-bold uppercase tracking-tighter leading-none">Security Protocol</span>
              </div>
              <p className="text-[9px] text-slate-500 font-mono uppercase leading-tight italic">Containment failure is not an option. Monitor all data streams.</p>
            </div>
          </div>
        </aside>

        {viewMode === 'browse' ? (
          <>
            <div className="w-full md:w-80 bg-white border-r border-slate-200 overflow-y-auto flex flex-col">
              <div className="p-3 border-b border-slate-100 bg-slate-50 flex justify-between items-center sticky top-0 z-10">
                <span className="text-[10px] font-bold text-slate-400 uppercase tracking-widest font-mono italic">Asset Stream</span>
                <span className="text-[10px] font-mono text-slate-400">[{filteredDocs.length}]</span>
              </div>
              <div className="divide-y divide-slate-100">
                {filteredDocs.map(doc => (
                  <div 
                    key={doc.id}
                    onClick={() => setSelectedDoc(doc)}
                    className={`p-4 cursor-pointer transition group border-l-4 ${selectedDoc?.id === doc.id ? 'border-amber-600 bg-amber-50 shadow-inner' : 'border-transparent hover:bg-slate-50'}`}
                  >
                    <div className="flex items-center justify-between mb-1">
                      <div className="flex items-center space-x-2">
                        {doc.type === 'text' && <FileText className="w-3 h-3 text-slate-400" />}
                        {doc.type === 'image' && <ImageIcon className="w-3 h-3 text-slate-400" />}
                        {doc.type === 'pdf' && <FileCode className="w-3 h-3 text-amber-600" />}
                        <span className="text-[9px] font-bold text-slate-500 uppercase tracking-tighter">{doc.category}</span>
                      </div>
                    </div>
                    <div className="flex justify-between items-start">
                      <h3 className={`text-sm font-bold leading-tight ${selectedDoc?.id === doc.id ? 'text-amber-900' : 'text-slate-800'}`}>{doc.title}</h3>
                      {canDelete && (
                        <div className="relative">
                          {purgeConfirmId === doc.id ? (
                             <button onClick={(e) => { e.stopPropagation(); handleDelete(doc.id); }} className="text-red-600 animate-bounce p-1"><ShieldCheck className="w-4 h-4" /></button>
                          ) : (
                            <button onClick={(e) => { e.stopPropagation(); setPurgeConfirmId(doc.id); }} className="text-slate-300 hover:text-red-600 transition p-1 opacity-0 group-hover:opacity-100"><Trash2 className="w-3 h-3" /></button>
                          )}
                        </div>
                      )}
                    </div>
                  </div>
                ))}
              </div>
            </div>

            <div className="flex-1 bg-slate-200 p-8 overflow-y-auto flex flex-col items-center">
              {selectedDoc ? (
                <div className="w-full max-w-4xl bg-white shadow-2xl border border-slate-300 min-h-full flex flex-col text-left rounded-sm">
                  <div className="p-6 border-b-2 border-slate-100 bg-slate-50 flex justify-between items-center">
                    <div>
                      <h2 className="text-xl font-black text-slate-900 uppercase tracking-tighter leading-none">{selectedDoc.title}</h2>
                      <p className="text-[10px] font-mono text-slate-500 uppercase mt-2">INDEX DATE: {selectedDoc.dateAdded} // UID: {selectedDoc.id.slice(0,12)}</p>
                    </div>
                    <div className="bg-slate-900 text-amber-500 text-[10px] font-mono font-bold px-3 py-1 rounded-sm border border-amber-900 uppercase tracking-widest shadow-lg">
                      {selectedDoc.classification}
                    </div>
                  </div>
                  
                  <div className="p-8 flex-1">
                    {selectedDoc.type === 'text' && (
                      <div className="font-mono text-sm text-slate-700 whitespace-pre-wrap leading-relaxed border-l-4 border-amber-500 pl-8 bg-slate-50 p-8 rounded-sm shadow-inner italic">
                        {selectedDoc.content}
                      </div>
                    )}
                    
                    {selectedDoc.type === 'image' && (
                      <div className="flex flex-col items-center">
                        <div className="bg-slate-900 p-2 rounded shadow-2xl max-w-full inline-block border border-slate-700">
                           <img src={selectedDoc.fileData} className="max-w-full rounded-sm" alt="Lore Evidence" />
                        </div>
                        <p className="mt-8 text-[10px] font-mono text-slate-400 uppercase tracking-[0.3em] italic border-t border-slate-100 pt-4 w-full text-center">Visual Uplink Confirmed</p>
                      </div>
                    )}

                    {selectedDoc.type === 'pdf' && (
                      <div className="w-full h-[750px] bg-slate-800 border-4 border-slate-300 rounded shadow-2xl overflow-hidden">
                        <iframe src={selectedDoc.fileData} className="w-full h-full" title="PDF Document" />
                      </div>
                    )}
                  </div>
                </div>
              ) : (
                <div className="h-full flex flex-col items-center justify-center text-slate-400 space-y-4 opacity-10 font-mono text-sm uppercase tracking-[0.5em]">
                  <Database className="w-24 h-24 mb-4 text-slate-400" />
                  <p>InGen Archive Terminal Offline</p>
                </div>
              )}
            </div>
          </>
        ) : (
          <div className="flex-1 bg-slate-100 p-12 overflow-y-auto">
            <div className="max-w-3xl mx-auto bg-white border border-slate-300 shadow-sm p-12 relative overflow-hidden rounded-sm">
              <div className="absolute top-0 right-0 p-4 opacity-5"><Database className="w-64 h-64 -mr-16 -mt-16" /></div>
              <div className="border-b-4 border-slate-900 pb-4 mb-8">
                <h2 className="text-3xl font-black text-slate-900 uppercase italic tracking-tighter">Archival Notice & Scope</h2>
                <p className="font-mono text-[10px] text-slate-500 font-bold uppercase tracking-[0.2em]">FILE: ALPHA-00-GENERAL</p>
              </div>
              <div className="space-y-8 text-slate-700 font-mono text-sm leading-relaxed">
                <section>
                  <h3 className="font-bold text-slate-900 uppercase border-b-2 border-amber-400 mb-2">I. Preservation Objective</h3>
                  <p>This terminal is a dedicated repository for the "lost" history of InGen, Jurassic Park, and broader prehistoric lore. We focus on rare evidence of genetic experiments, primal anomalies, and general Jurassic-related legends often removed from official records.</p>
                </section>
                <section>
                  <h3 className="font-bold text-slate-900 uppercase border-b-2 border-amber-400 mb-2">II. Unofficial Documentation</h3>
                  <p>This platform is for historical and creative purposes. We are **not affiliated with or endorsed by** NBCUniversal or Amblin Entertainment. This project is a fan tribute to the "Primal" world and the forgotten creations of cinema and theme park history.</p>
                </section>
                <section>
                  <h3 className="font-bold text-slate-900 uppercase border-b-2 border-amber-400 mb-2">III. Access Protocol</h3>
                  <p>Viewing access is public. Data ingestion is restricted to verified personnel with Security Override clearance to ensure archive integrity.</p>
                </section>
              </div>
            </div>
          </div>
        )}
      </div>

      {/* AUTHORIZATION MODAL */}
      {isUploadOpen && (
        <div className="fixed inset-0 bg-slate-900/90 backdrop-blur-sm flex items-center justify-center p-4 z-50">
          <div className="bg-white w-full max-w-lg shadow-[0_0_60px_rgba(217,119,6,0.15)] border-t-8 border-amber-600 rounded-sm overflow-hidden flex flex-col max-h-[90vh]">
            <div className="p-4 bg-slate-50 border-b flex justify-between items-center">
              <span className="font-mono text-xs font-bold uppercase text-slate-600 tracking-widest flex items-center gap-2">
                <Lock className={`w-3 h-3 ${isAdminSession ? 'text-amber-500' : 'text-slate-300'}`} /> 
                {isAdminSession ? "InGen Data Ingestion" : "Access Authorization"}
              </span>
              <button onClick={() => setIsUploadOpen(false)} className="text-slate-400 hover:text-slate-800 transition"><X /></button>
            </div>
            
            <div className="p-6 space-y-5 overflow-y-auto">
              {errorMsg && <div className="p-3 bg-red-50 text-red-700 text-[10px] font-mono border-l-4 border-red-500 flex items-center space-x-2">
                <AlertTriangle className="w-3 h-3" />
                <span>{errorMsg}</span>
              </div>}
              
              {!isAdminSession ? (
                <div className="space-y-6">
                   <div className="bg-amber-50 p-4 border border-amber-100 rounded text-[10px] font-mono text-amber-800 italic uppercase">
                     Mainframe Identity Verification Required. Credentials must be active for this sector.
                   </div>
                   <div className="space-y-4">
                      <div className="space-y-1">
                        <label className="text-[9px] font-bold text-slate-400 uppercase font-mono tracking-widest">Insert ID</label>
                        <input 
                          type="text" 
                          value={idInput} 
                          onChange={e => setIdInput(e.target.value)}
                          className="w-full p-3 border border-slate-300 rounded-sm text-sm focus:border-amber-600 outline-none font-mono bg-slate-50 shadow-inner" 
                          placeholder="DESIGNATION REQUIRED..." 
                        />
                      </div>
                      <div className="space-y-1">
                        <label className="text-[9px] font-bold text-slate-400 uppercase font-mono tracking-widest">Protocol Code</label>
                        <div className="relative">
                          <input 
                            type={showCode ? "text" : "password"} 
                            value={authInput} 
                            onChange={e => setAuthInput(e.target.value.toUpperCase())}
                            className="w-full p-3 border border-slate-300 rounded-sm text-sm focus:border-amber-600 outline-none font-mono bg-slate-50 shadow-inner tracking-[0.4em]" 
                            placeholder="••••"
                          />
                          <button onClick={() => setShowCode(!showCode)} className="absolute right-3 top-1/2 -translate-y-1/2 text-slate-300 hover:text-amber-600">
                            {showCode ? <EyeOff className="w-4 h-4" /> : <Eye className="w-4 h-4" />}
                          </button>
                        </div>
                      </div>
                   </div>
                   <button 
                    onClick={handleAuthorize}
                    className="w-full bg-slate-900 text-amber-500 py-4 font-bold uppercase tracking-[0.4em] text-xs hover:bg-black transition rounded-sm shadow-xl"
                   >
                     Authorize Uplink
                   </button>
                </div>
              ) : (
                <>
                  <div className="space-y-1">
                    <label className="text-[9px] font-bold text-slate-400 uppercase font-mono tracking-widest">Subject Name</label>
                    <input type="text" value={formData.title} onChange={e => setFormData({...formData, title: e.target.value})} className="w-full p-3 border border-slate-300 rounded-sm text-sm focus:border-amber-600 outline-none font-mono uppercase bg-slate-50 shadow-inner" placeholder="SUBJECT DESIGNATION..." />
                  </div>

                  <div className="grid grid-cols-2 gap-4">
                    <div className="space-y-1">
                      <label className="text-[9px] font-bold text-slate-400 uppercase font-mono tracking-widest">Directory</label>
                      <select value={formData.category} onChange={e => setFormData({...formData, category: e.target.value})} className="w-full p-2.5 border border-slate-300 rounded-sm text-sm focus:border-amber-600 outline-none font-mono bg-white uppercase">
                        <option>InGen Files</option>
                        <option>Primal Lore</option>
                        <option>HHN Legacy</option>
                        <option>Genetic Research</option>
                        <option>Forgotten History</option>
                      </select>
                    </div>
                    <div className="space-y-1">
                      <label className="text-[9px] font-bold text-slate-400 uppercase font-mono tracking-widest">Clearance</label>
                      <select value={formData.classification} onChange={e => setFormData({...formData, classification: e.target.value})} className="w-full p-2.5 border border-slate-300 rounded-sm text-sm focus:border-amber-600 outline-none font-mono bg-white uppercase font-bold text-amber-600">
                        <option>Restricted</option>
                        <option>Confidential</option>
                        <option>Classified</option>
                        <option>Bio-Hazard</option>
                      </select>
                    </div>
                  </div>

                  <div className="space-y-1">
                    <div className="flex space-x-1">
                      {['text', 'image', 'pdf'].map(type => (
                        <button key={type} onClick={() => { setFormData({...formData, type}); setSelectedFile(null); }} className={`flex-1 py-2 text-[9px] font-bold uppercase rounded-sm border-2 transition font-mono ${formData.type === type ? 'bg-slate-900 border-slate-900 text-amber-500 shadow-md' : 'bg-slate-50 border-slate-200 text-slate-400'}`}>
                          {type}
                        </button>
                      ))}
                    </div>
                  </div>

                  {formData.type === 'text' ? (
                    <textarea rows={5} value={formData.content} onChange={e => setFormData({...formData, content: e.target.value})} className="w-full p-3 border border-slate-300 rounded-sm text-sm font-mono focus:border-amber-600 outline-none bg-slate-50 shadow-inner" placeholder="INPUT RAW LORE DATA..." />
                  ) : (
                    <div className="border-2 border-dashed border-slate-300 p-8 rounded text-center bg-slate-50 hover:bg-amber-50/50 transition">
                      <input type="file" id="file" className="hidden" accept={formData.type === 'image' ? 'image/*' : 'application/pdf'} onChange={e => setSelectedFile(e.target.files[0])} />
                      <label htmlFor="file" className="cursor-pointer">
                        <Upload className="w-8 h-8 text-amber-400 mx-auto mb-2" />
                        <p className="text-[10px] text-slate-600 font-mono font-bold uppercase tracking-widest">{selectedFile ? selectedFile.name : `Select ${formData.type.toUpperCase()} Source`}</p>
                      </label>
                    </div>
                  )}

                  <button 
                    onClick={handleUpload} 
                    disabled={isUploading || !formData.title} 
                    className="w-full bg-slate-900 text-amber-500 py-4 font-bold uppercase tracking-[0.4em] text-xs hover:bg-black transition flex items-center justify-center space-x-2 rounded-sm disabled:opacity-50 shadow-2xl"
                  >
                    {isUploading ? <><Loader2 className="w-4 h-4 animate-spin" /> <span>Syncing...</span></> : <span>Commit Data</span>}
                  </button>
                  <button 
                    onClick={() => setIsAdminSession(false)} 
                    className="w-full text-[9px] font-mono text-slate-400 uppercase hover:text-red-500 transition tracking-widest pt-2"
                  >
                    Terminate Authorized Session
                  </button>
                </>
              )}
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
